# 验证日期 DatePicker

<Form ref="formInline" :model="formItem" :rules="ruleInline" inline>
  <FormItem prop="Time">
    <DatePicker :editable="false" v-model="formItem.Time" type="date" format="yyyy-MM-dd" placeholder="请选择时间" ></DatePicker>
  </FormItem>
</Form>

data() {
  const validateTime = (rule, value, callback) => {
    if (this.formItem.Time == '') {
      callback(new Error('请选择有效时间'))
    }else {
      callback()
    }
  };

  return {
    formItem: {
      Time: ''
    },
    ruleInline: {
      Time:[
        { validator: validateTime, trigger: 'change' }
      ]
    }
  }
},
methods: {
  // 提交基本信息
  submitBaseInfo(){
    this.$refs['formInline'].validate((valid) => {
      if(valid){
        console.log('验证成功')
      } else {
        this.$Message.error('请填写完整信息');
      }
    })
  }
}
