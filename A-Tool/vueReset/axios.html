# 安装

npm install axios --save-dev

# 配置

main.js中

import axios from 'axios'
Vue.prototype.$axios = axios

# 跨域

根目录新建vue.config.js

module.exports = {
  devServer: {
    open: true,
    host: 'localhost',
    port: 8080,
    https: false,
    hotOnly: false,
    proxy: {
      '/api': {
        target: 'localhost:3000/api/',
        ws: true,
        changOrigin: true,
        pathRewrite: {
          '^/api': ''
        }
      }
    }
  }
}
