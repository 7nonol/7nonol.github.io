# 配置

html

<ul class="nav flex">
  <li
    class="nav-list"
    :class="navIndex == index ? 'active' : ''"
    @click.capture="navSelected(index, item)"
    @mouseenter="enter(index)"
    @mouseleave="leave(index)"
    v-for="(item, index) in configNav"
    :key="index"
  >
    <router-link :to="item.path" class="nav-list-btn flex ac">
      {{ item.name }}
    </router-link>

    <ul ref="subnav" class="subnav">
      <li
        class="subnav-list"
        v-for="(subnav, i) in item.subItems"
        :class="subnavIndex == i ? 'active' : ''"
        @click="subnavSelected(i, index)"
        :key="i"
      >
        <router-link class="subnav-list-btn" :to="subnav.link">{{
          subnav.text
        }}</router-link>
      </li>
    </ul>
  </li>
</ul>

js

data() {
  return {
    navIndex: 0,
    subnavIndex: -1,
    configNav: [
      {
        name: '首页',
        path: '/home'
      },
      {
        name: '新闻',
        path: '',
        subItems: [
          { link: '/new-pay', text: '财经新闻' },
          { link: '/new-happy', text: '娱乐新闻' }
        ]
      },
      {
        name: 'PXxx',
        path: ''
      },
      {
        name: '关于我们',
        path: ''
      }
    ]
  }
},
methods: {
  // 一级导航添加选中样式
  navSelected: function(index, item) {
    if (!item.subItems) {
      this.navIndex = index
      this.subnavIndex = -1
    }
  },

  // 二级导航添加选中样式
  subnavSelected: function(i, index) {
    this.subnavIndex = i
    this.navIndex = index
    this.$refs.subnav[index].style.display = 'none'
  },

  // 监听一级导航悬停显示
  enter(index) {
    this.$refs.subnav[index].style.display = 'block'
  },

  // 监听一级导航悬停隐藏
  leave(index) {
    this.$refs.subnav[index].style.display = 'none'
  }
}
