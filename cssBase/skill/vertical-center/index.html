# 未知父元素高度

// 方法1
.parentDom {
  position: relative;
}
.childDom {
  position: absolute;
  top: 50%;
  transform: translateY(-50%);
}

// 方法2
.parentDom {
  position: relative;
}
.childDom {
  position: absolute;
  top: 0;
  bottom: 0;
  margin: auto;
}
