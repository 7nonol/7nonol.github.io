methods: {
  // 点击日期
  changeDate(date) {
    if(date[0]){
      this.searchParams.criteria = 'Custom'
      this.searchParams.startTime = new Date(date[0] + ' 00:00:00')
      this.searchParams.endTime = new Date(date[1] + ' 23:59:59')
    } else {
      this.searchParams.criteria = ''
      this.searchParams.startTime = ''
      this.searchParams.endTime = ''
    }
  }
}

# DatePicker events

事件名 说明 返回值
on-change 日期发生变化时触发 返回两个值，已经格式化后的日期，比如 2016-01-01，和当前的日期类型，比如 date
on-clear 在 confirm 模式或 clearable = true 时有效，在清空日期时触发 -
