data() {
  return {
    loading: true,
    searchParams: {
      // 查询条件
      pageNum: 1,
      pageSize: 10,
      total: 0
    },
    columns: [
      {
        title: 'Name',
        key: 'name',
        minWidth: 120,
        align: 'center'
      },
      {
        title: 'Age',
        key: 'age',
        minWidth: 120,
        align: 'center'
      },
      {
        ...this.opColumn()
      }
    ],
    list: []
  }
},
methods: {
  // 获取数据
  getList() {
    storeList(this.searchParams).then(res => {
      if (res.data.code == 0) {
        this.loading = false
        this.list = res.data.data.list
        this.searchParams.total = res.data.data.total
      }
    })
  },

  // 点击分页
  changePage(num) {
    this.searchParams.pageNum = num
    this.getList()
  },

  // 操作按钮
  opColumn() {
    let colwidth = this.icons.length * 45
    colwidth = colwidth < 100 && colwidth !== 0 ? 100 : colwidth
    colwidth = colwidth === 0 ? 1 : colwidth // 隐藏操作列
    if (this.searchParams.state == -1) {
      colwidth = 1
    }
    let col = {
      title: '操作',
      key: 'action',
      align: 'center',
      fixed: 'right',
      width: colwidth,
      render: (h, params) => {
        const showIcon = h(TooltipIcon, {
          props: { content: '申请详情', fontIcon: 'ios-paper-outline' },
          on: {
            click: () => {
              this.rowClickEvent('VIEW', '评论详情', params.row)
            }
          }
        })
        const setIcon = h(TooltipIcon, {
          props: {
            content: params.row.state == 0 ? '禁用' : '启用',
            fontIcon:
              params.row.state == 0
                ? 'ios-close-circle-outline'
                : 'ios-checkmark-circle-outline',
            color: params.row.state == 0 ? 'red' : '#000'
          },
          on: {
            click: () => {
              this.rowClickEvent('SET', '操作', params.row)
            }
          }
        })
        let re = []
        // ['DTL', 'MOVE', 'CHG', 'DEL']
        if (this.icons.indexOf('VIEW') > -1) {
          re.push(showIcon)
        }
        if (this.icons.indexOf('SET') > -1) {
          re.push(setIcon)
        }
        return re
      }
    }
    return col
  }
}
