import { promoterList } from '@/api/member.js'

props: {
  searchParams: {
    type: Object
  },
  btns: {
    type: Array,
    default: function() {
      return []
    }
  },
  refresh:{
    default: false,
    type: Boolean
  }
},
data() {
  return {
    columns: [
      {
        title: "表头一",
        key: "name1",
        minWidth:120,
        align: 'center'
      },
      {
        title: "表头二",
        key: "name2",
        minWidth:120,
        align: 'center'
      },
      {
        ...this.opColumn()
      }
    ],
    list: [],
    loading: false
  }
},
watch:{
  refresh() {
    this.getList()
  }
},
methods: {
  // 列表数据
  getList() {
    this.loading = true
    promoterList(this.searchParams).then(res => {
      console.log(res,1111)
      if(res.data.code == 0){
        this.list = res.data.page.list
        this.searchParams.total = res.data.page.total
        this.loading = false
      }else{
        this.loading = false;
      }
    }).catch(err => {
      this.loading = false
    })
  },

  // 操作按钮
  opColumn(){
    let colwidth = this.btns.length * 64
    colwidth = (colwidth < 100 && colwidth !== 0) ? 100 : colwidth
    colwidth = colwidth === 0 ? 1 : colwidth // 隐藏操作列
    let col = {
      title: '操作',
      key: 'action',
      align: 'center',
      fixed: 'right',
      width: colwidth,
      render:(h, params) => {
        const viewBtn = h('Button',
          { 
            'props': {
              'type': 'primary',
              'size': 'small'
            },
            'style': {
              'margin': '0 3px 0 3px'
            },
            on: {
              click: () => {
                this.rowClickEvent('VIEW', '查看', params.row)
              }
            }
          }, '查看'
        )
        const setBtn = h('Button',
          { 
            'props': {
              'type': 'primary',
              'size': 'small'
            },
            'style': {
              'margin': '0 3px 0 3px'
            },
            on: {
              click: () => {
                this.rowClickEvent('SET', '查看', params.row)
              }
            }
          }, '设置'
        )
        const delBtn = h('Button',
          { 
            'props': {
              'type': 'error',
              'size': 'small'
            },
            'style': {
              'margin': '0 3px 0 3px'
            },
            on: {
              click: () => {
                this.rowClickEvent('DEL', '禁用', params.row)
              }
            }
          }, '禁用'
        )
        let re = []
        if (this.btns.indexOf('VIEW') > -1) {
          re.push(viewBtn)
        }
        if (this.btns.indexOf('DEL') > -1) {
          re.push(delBtn)
        }
        if (this.btns.indexOf('SET') > -1) {
          re.push(setBtn)
        }
        return re
      }
    }
    return col
  },

  // 列表数据
  getList() {
    this.loading = true
    promoterList(this.searchParams).then(res => {
      if(res.data.code == 0){
        this.list = res.data.page.list
        this.searchParams.total = res.data.page.total
        this.loading = false
      }else{
        this.loading = false
      }
    }).catch(err => {
      this.loading = false
    })
  },

  // 点击按钮 操作
  rowClickEvent(action, actionDesc, item) {
    this.$emit('on-click', action, actionDesc, item)
  },
  
  // 点击分页
  changePage(num) {
    this.searchParams.pageNum = num
    this.getList()
  }
}
