# addOrEditMoadl.vue

<template>
  <Modal v-model="showModal" title="设置状态">
    <Form ref="formInline" :model="formInline" :rules="ruleInline">
      <FormItem label="状态">
        <Select v-model="rewardInfo.reviewState" placeholder="请选择列表状态" clearable>
          <Option v-for="item in selectList" :value="item.value" :key="item.value">{{ item.label }}</Option>
        </Select>
      </FormItem>
    </Form>
    <div slot="footer">
      <Button @click="showModal=false">取消</Button>
      <Button type="primary" :loading="loading" @click="handleSubmit('formInline')">确定</Button>
    </div>
  </Modal>
</template>

<script>
import { mapState } from "vuex";

export default {
  props: {
    value: {
      type: Boolean
    }
  },
  data() {
    return {
      formInline: {

      },
      ruleInline: {
        
      },
      selectList: [
        {
          value: "1",
          label: "审核成功"
        },
        {
          value: "-1",
          label: "审核失败"
        }
      ],
      loading: false
    };
  },
  computed: {
    ...mapState(["rewardInfo"]),
    showModal: {
      get(val) {
        return this.value;
      },
      set(val) {
        this.$emit("input", val);
      }
    }
  },
  methods: {
    // 验证表单
    handleSubmit(name) {
      this.add();
    },

    // 提交 添加
    add() {
      let params = {
        id: this.rewardInfo.id,
        state: this.rewardInfo.reviewState,
      }
      this.loading = true;
      this.$http
        .json_post(this.$api.promoteUrl.changeRewardList, params)
        .then(data => {
          if (data.code == 0) {
            this.$Message.success("设置成功");
            this.showModal = false;
            this.loading = false;
            this.$emit('on-success')
          } else {
            this.loading = false;
          }
        })
        .catch(err => {
          this.loading = false;
        });
    }
  }
};
</script>
