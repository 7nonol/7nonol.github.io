import TableTemplate from '../tableComponents/table-template'

components: {
  TableTemplate
},

  return {
    searchParams: {// 搜索条件
      pageNum: 1,
      pageSize: 10,
      total: 0,
      inputVal: '',
      selectVal: ''
    },
    selectList: [
      {
        value: '0',
        label: '下拉0'
      },
      {
        value: '1',
        label: '下拉1'
      }
    ],
    refresh: false,
  }
},
computed:{
  ...mapState(['extensionInfo'])
},
mounted() {
  this.refresh = !this.refresh
},
methods: {
  // table按钮操作
  rowClickEvent(action, actionDesc, item) {
    if (action === 'VIEW') {
      alert('查看')
    } else if (action === 'MOD') {
      alert('禁用/开启')
    }
  },

  // 搜索
  search() {
    this.searchParams.pageNum = 1
    this.$refs.tabelChild.getList()
  }
}

import TooltipIcon from '_c/tooltipIcon'
import { publicFn } from '@/utils'
import { mapMutations } from 'vuex'

props: {
  searchParams: {
    type: Object
  },
  icons: {
    type: Array,
    default: function() {
      return []
    }
  },
  refresh:{
    default: false,
    type: Boolean
  }
},
data() {
  return {
    columns: [
      {
        title: "表头一",
        key: "name1",
        align: 'center'
      },
      {
        title: "表头二",
        key: "name2",
        align: 'center'
      },
      {
        ...this.opColumn()
      }
    ],
    list: [],
    loading: false
  }
},
watch:{
  refresh() {
    this.getList()
  }
},
methods: {
  ...mapMutations(['SET_EXTENSION']),

  // 列表数据
  getList() {
		this.loading = true
		this.$http.json_post(this.$api.userUrl.extensionList, this.searchParams).then(data => {
			if (data.code == 0) {
				this.list = data.page.list
				this.searchParams.total = data.page.total
				this.loading = false
			} else {
				this.loading = false
			}
		}).catch(err => {
      this.loading = false
    })
  },

  // 操作按钮
  opColumn(){
		let colwidth = this.icons.length * 45
		colwidth = (colwidth < 100 && colwidth !== 0) ? 100 : colwidth
		colwidth = colwidth === 0 ? 1 : colwidth // 隐藏操作列
		let col = {
			title: '操作',
			key: 'action',
			align: 'center',
			fixed: 'right',
			width: colwidth,
			render: (h, params) => {
				const editIcon = h(TooltipIcon,
					{
						props: { content: '编辑', fontIcon: 'ios-settings-outline' },
						on: {
							click: () => {
								this.rowClickEvent('EDIT', '编辑', params.row)
							}
						}
					}
				)
				const downUeIcon = h(TooltipIcon,
					{
						props: { content: '下载邀请用户二维码', fontIcon: 'ios-cloud-download-outline' },
						on: {
							click: () => {
								window.open(this.$api.userUrl.downLoad+ params.row.uextensionQrCode)
							}
						}
					}
				)
				const downBeIcon = h(TooltipIcon,
					{
						props: { content: '下载邀请商户二维码', fontIcon: 'ios-cloud-download-outline' },
						on: {
							click: () => {
								window.open(this.$api.userUrl.downLoad+ params.row.bextensionQrCode)
							}
						}
					}
				)
				const detailIcon = h(TooltipIcon,
					{
						props: { content: '用户详情', fontIcon: 'ios-paper-outline' },
						on: {
							click: () => {
								this.rowClickEvent('VIEW', '用户详情', params.row)
							}
						}
					}
				)
				const modIcon = h(TooltipIcon,
					{
						props: { content: params.row.state == 0 ? '禁用' : '启用', fontIcon: 'ios-alert-outline' },
						on: {
							click: () => {
								this.rowClickEvent('MOD', params.row.state == 0 ? '禁用' : '启用', params.row)
							}
						}
					}
				)

				let re = []
				if (this.icons.indexOf('EDIT') > -1) {
					re.push(editIcon)
				}
				if (this.icons.indexOf('DETAIL') > -1) {
					re.push(detailIcon)
				}
				if (this.icons.indexOf('MOD') > -1) {
					re.push(modIcon)
				}
				return re
			}
		}
		return col
	},

  // 点击按钮 操作
  rowClickEvent(action, actionDesc, item) {
    this.SET_EXTENSION(item)
    this.$emit('on-click', action, actionDesc, item)
  },
  
  // 点击分页
  changePage(num) {
    this.searchParams.pageNum = num
    this.getList()
  }
}
