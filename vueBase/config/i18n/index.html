# 配置

# 文件结构

# 1.tools文件夹中新建i18n.js文件(使项目结构清晰)

import Vue from 'vue'
import VueI18n from 'vue-i18n'
Vue.use(VueI18n)

const messages = {
  zh_CN: require('../assets/i18n/zh'), // 中文语言包
  en_US: require('../assets/i18n/en') // 英文语言包
}

export default new VueI18n({
  locale: 'zh_CN', // set locale 默认显示英文
  messages: messages // set locale messages
})

# 2.在main.js里面全局引入并挂载

import i18n from './tools/i18n'

new Vue({
  i18n,
  router,
  store,
  render: h => h(App)
}).$mount('#app')

# 3. ../assets/i18n/zh 和 ../assets/i18n/en

zh.js

module.exports = {
  login: {
    username: '请输入用户名',
    password : '请输入密码',
  }
}

en.js

module.exports = {
  login: {
    username: 'Please enter the user name',
    password : 'Please enter your password',
  }
}

# 4.login.vue调用

<h3>{{ $t("login.username") }}</h3>
<input type="text" :placeholder="$t("login.password")">
<button @click='langToggle'>en/中</button>

<script>
methods:{
  langToggle() {
    if ((this.$i18n.locale = 'zh_CN')) {
      this.$i18n.locale = 'en_US'
    } else {
      this.$i18n.locale = 'zh_CN'
    }
  }
}
</script>
