# $router和$route的区别

  • $router : 路由操作对象，只写对象
  • $route :  路由信息对象，只读对象
// 操作 路由跳转
this.$router.push({
  name: 'hello',
  params: {
    name: 'word',
    age: '11'
  }
})

//读取 路由参数接收
this.name = this.$route.params.name
this.age = this.$route.params.age
